    `CopilotSidebar` wraps `CopilotChat` in a collapsible sidebar.

    First, import the default styles in your root component:

    ```tsx
    import "@copilotkit/react-ui/styles.css";
    ```

    Then, use the component in your application:

    ```tsx
    import { CopilotSidebar } from "@copilotkit/react-ui";

    export function YourApp() {
      return (
        <CopilotSidebar
          defaultOpen={true}
          labels={{
            title: "Sidebar Assistant",
            initial: "How can I help you today?",
          }}
        >
          <YourMainContent />
        </CopilotSidebar>
      );
    }
    ```

    Key properties:
    - `defaultOpen`: Whether the sidebar is open by default
    - `clickOutsideToClose`: Close sidebar when clicking outside
    - `shortcut`: Keyboard shortcut to toggle sidebar

    `CopilotPopup` wraps `CopilotChat` in a toggleable popup window.

    First, import the default styles in your root component:

    ```tsx
    import "@copilotkit/react-ui/styles.css";
    ```

    Then, use the component in your application:

    ```tsx
    import { CopilotPopup } from "@copilotkit/react-ui";

    export function YourApp() {
      return (
        <>
          <YourMainContent />
          <CopilotPopup
            labels={{
              title: "Popup Assistant",
              initial: "Need any help?",
            }}
          />
        </>
      );
    }
    ```

    Key properties:
    - `defaultOpen`: Whether the popup is open by default
    - `hitEscapeToClose`: Close popup when pressing Escape key
    - `shortcut`: Keyboard shortcut to toggle popup

    For custom UI implementations, use the `useCopilotChat` hook.

    ```tsx
    import { useCopilotChat } from "@copilotkit/react-core";
    import { Role, TextMessage } from "@copilotkit/runtime-client-gql";

    export function CustomChatInterface() {
      const {
        visibleMessages,
        appendMessage,
        setMessages,
        deleteMessage,
        reloadMessages,
        stopGeneration,
        isLoading,
      } = useCopilotChat();

      const sendMessage = (content: string) => {
        appendMessage(
          new TextMessage({
            content,
            role: Role.User,
          })
        );
      };

      return (
        <div>
          {/* Implement your custom chat UI here */}
        </div>
      );
    }
    ```

    - Full control over UI implementation
    - Access to chat state and operations
    - Customizable message handling
